300def urljoin(url, suffix=""): 

301 """ 

302 Will join url and its suffix 

303 

304 Example: 

305 "https://google.com/", "/" => "https://google.com/" 

306 "https://google.com", "/" => "https://google.com/" 

307 "https://google.com", "api" => "https://google.com/api" 

308 "https://google.com", "/api" => "https://google.com/api" 

309 "https://google.com/", "api" => "https://google.com/api" 

310 "https://google.com/", "/api" => "https://google.com/api" 

311 

312 :type url: ``string`` 

313 :param url: URL string (required) 

314 

315 :type suffix: ``string`` 

316 :param suffix: the second part of the url 

317 

318 :rtype: ``string`` 

319 :return: Full joined url 

320 """ 

321 if url[-1:] != "/": 

322 url = url + "/" 

323 

324 if suffix.startswith("/"): 

325 suffix = suffix[1:] 

326 return url + suffix 

327 

328 return url + suffix

599 """ 

600 execute_command_result must contain error entry - check the result first with is_error function 

601 if there is no error entry in the result then it will raise an Exception 

602

605 

606 :return: Error message extracted from the demisto.executeCommand() result 

607 :rtype: ``string`` 

608 """ 

609 

610 if not is_error(execute_command_result): 

611 raise ValueError("execute_command_result has no error entry. before using get_error use is_error") 

612 

613 if isinstance(execute_command_result, dict): 

614 return execute_command_result['Contents'] 

615 

616 error_messages = [] 

617 for entry in execute_command_result: 

618 is_error_entry = type(entry) == dict and entry['Type'] == entryTypes['error'] 

619 if is_error_entry: 

620 error_messages.append(entry['Contents']) 

621 

622 return '\n'.join(error_messages)

1559def elem_to_internal(elem, strip_ns=1, strip=1): 

1560 """Convert an Element into an internal dictionary (not JSON!).""" 

1561 

1562 d = OrderedDict() # type: dict 

1563 elem_tag = elem.tag 

1564 if strip_ns: 

1565 elem_tag = strip_tag(elem.tag) 

1566 for key, value in list(elem.attrib.items()): 

1567 d['@' + key] = value 

1568 

1569 # loop over subelements to merge them 

1570 for subelem in elem: 

1571 v = elem_to_internal(subelem, strip_ns=strip_ns, strip=strip) 

1572 

1573 tag = subelem.tag 

1574 if strip_ns: 

1575 tag = strip_tag(subelem.tag) 

1576 

1577 value = v[tag] 

1578 try: 

1579 # add to existing list for this tag 

1580 d[tag].append(value) 

1581 except AttributeError: 

1582 # turn existing entry into a list 

1583 d[tag] = [d[tag], value] 

1584 except KeyError: 

1585 # add a new non-list entry 

1586 d[tag] = value 

1587 

1588 text = elem.text 

1589 tail = elem.tail 

1590 if strip: 

1591 # ignore leading and trailing whitespace 

1592 if text: 

1593 text = text.strip() 

1594 if tail: 

1595 tail = tail.strip() 

1596 

1597 if tail: 

1598 d['#tail'] = tail 

1599 

1600 if d: 

1601 # use #text element if other attributes exist 

1602 if text: 

1603 d["#text"] = text 

1604 else: 

1605 # text is the value if no attributes 

1606 d = text or None # type: ignore 

1607 return {elem_tag: d}

1610def internal_to_elem(pfsh, factory=ET.Element):

1641 e = factory(tag, attribs) 

1642 for sub in sublist: 

1643 e.append(sub) 

1644 e.text = text 

1645 e.tail = tail
